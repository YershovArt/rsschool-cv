Using a recursive function r5 replace tags inside z-5 that have no nested tags with mark tags. 

					
let ar7 = [
		 'hi',
		 ['hii'],
		 [['hiii']],
		 [[['hiiii']]],
		 ['i', [[[[[['hi']]]]]]]
	];
	let ar7_res = [];
	
	function t7(arg) {
		 if (Array.isArray(arg)) {
			  arg.forEach(element => {
					if (typeof element === 'string') ar7_res.push(element);
					else t7(element)
			  });
		 }
	}
	
	document.querySelector('.b-7').addEventListener('click', () => {
		 t7(ar7);
		 document.querySelector('.out-7').textContent = ar7_res;
	});

Using a recursive function r5 replace tags inside z-5 that have no nested tags with mark tags. 

					
function t5(elem) {
	r5(elem);
					  
	function r5(element) {
			if (element.children.length === 0) {
				let mark = document.createElement('mark');
				mark.textContent = element.textContent;
				element.parentNode.appendChild(mark);
				element.parentNode.removeChild(element);
			}
			[...element.children].forEach(item => r5(item));
	}
}
					  
document.querySelector('.b-5').addEventListener('click', () => {
	let z5 = document.querySelector('.z-5');
	t5(z5);
});
